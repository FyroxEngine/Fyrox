App.tsx
import React, { useState, useCallback } from 'react';
import { Header } from './components/Header';
import { UniverseGenerator } from './components/UniverseGenerator';
import { ResultDisplay } from './components/ResultDisplay';
import { generateUniverseDetails } from './services/geminiService';
import { OriginStory, UniverseParameters, SimulationResult } from './types';

const App: React.FC = () => {
  const [simulationResult, setSimulationResult] = useState<SimulationResult | null>(null);
  const [isLoading, setIsLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  const handleSimulate = useCallback(async (originStory: OriginStory, params: UniverseParameters) => {
    setIsLoading(true);
    setError(null);
    setSimulationResult(null);

    try {
      const result = await generateUniverseDetails(originStory, params);
      setSimulationResult(result);
    } catch (err) {
      console.error("Simulation Error:", err);
      setError(err instanceof Error ? err.message : "An unknown error occurred during simulation.");
    } finally {
      setIsLoading(false);
    }
  }, []);

  return (
    <div className="min-h-screen bg-slate-900 text-slate-200 bg-grid-slate-700/[0.2]">
      <Header />
      <main className="container mx-auto px-4 py-8">
        <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">
          <div className="lg:col-span-4">
            <UniverseGenerator onSimulate={handleSimulate} isLoading={isLoading} />
          </div>
          <div className="lg:col-span-8">
            <ResultDisplay
              result={simulationResult}
              isLoading={isLoading}
              error={error}
            />
          </div>
        </div>
      </main>
      <footer className="text-center py-4 text-slate-500 text-sm">
        <p>BIOSPHERES Procedural Universe Engine</p>
      </footer>
    </div>
  );
};

export default App;

import { GoogleGenAI, Type } from "@google/genai";
import type { OriginStory, UniverseParameters, SimulationResult } from '../types';

if (!process.env.API_KEY) {
    throw new Error("API_KEY environment variable not set");
}

const ai = new GoogleGenAI({ apiKey: process.env.API_KEY });

const responseSchema = {
  type: Type.OBJECT,
  properties: {
    description: {
      type: Type.STRING,
      description: "A detailed, engaging, and imaginative narrative describing the universe created based on the user's parameters. This should be about 3-4 paragraphs long."
    },
    periodicTable: {
      type: Type.ARRAY,
      description: "A unique periodic table of 5-7 fictional elements that fit the theme of the origin story.",
      items: {
        type: Type.OBJECT,
        properties: {
          name: { type: Type.STRING, description: "The full name of the element." },
          symbol: { type: Type.STRING, description: "A 1-3 letter symbol for the element." },
          atomicNumber: { type: Type.INTEGER, description: "A unique atomic number for the element." },
          description: { type: Type.STRING, description: "A brief, imaginative description of the element's properties and role in the universe." },
          visualStyle: { type: Type.STRING, description: "A short text description of a Tailwind CSS class for a colored glow effect, e.g., 'shadow-cyan-500/50' or 'shadow-fuchsia-500/50'." },
        },
        required: ["name", "symbol", "atomicNumber", "description", "visualStyle"],
      },
    },
  },
  required: ["description", "periodicTable"],
};

export const generateUniverseDetails = async (
    originStory: OriginStory,
    params: UniverseParameters
): Promise<SimulationResult> => {
    
    const prompt = `
        You are the BIOSPHERES simulation engine. Your task is to generate the details of a new universe based on the specified parameters.

        **Origin Story:** ${originStory}
        
        **Universe Parameters:**
        - Dark Matter: ${params.darkMatter}%
        - AGN Feedback: Level ${params.agnFeedback}
        - Low Mass Stars: ${params.lowMassStars}%
        - Medium Mass Stars: ${params.mediumMassStars}%
        - High Mass Stars: ${params.highMassStars}%

        **Instructions:**
        1.  Write a compelling and imaginative narrative (3-4 paragraphs) describing the universe that would result from these parameters. The narrative should reflect the chosen origin story. For example, a 'Magical' universe might describe stars emitting magical energy, while a 'Mechanical Genesis' universe might have geometrically perfect galaxies.
        2.  Create a unique, fictional periodic table with 5-7 elements that logically fit the universe's theme. The properties and names of these elements should be creative and directly inspired by the origin story.
        3.  Provide the entire output in a single JSON object matching the provided schema.
    `;

    try {
        const response = await ai.models.generateContent({
            model: "gemini-2.5-flash",
            contents: prompt,
            config: {
                responseMimeType: "application/json",
                responseSchema: responseSchema,
                temperature: 0.8,
            }
        });

        const jsonText = response.text.trim();
        const result = JSON.parse(jsonText);

        // Basic validation
        if (!result.description || !Array.isArray(result.periodicTable)) {
            throw new Error("Invalid response format from API.");
        }

        return result as SimulationResult;

    } catch (error) {
        console.error("Error calling Gemini API:", error);
        throw new Error("Failed to generate universe details. Please check your API key and try again.");
    }
};

import React, { useState } from 'react';

import { SparklesIcon, ChevronDownIcon } from './Icons';

interface UniverseGeneratorProps {
  onSimulate: (originStory: OriginStory, params: UniverseParameters) => void;
  isLoading: boolean;
}

export const UniverseGenerator: React.FC<UniverseGeneratorProps> = ({ onSimulate, isLoading }) => {
  const [originStory, setOriginStory] = useState<OriginStory>(OriginStory.BIG_BANG);
  const [params, setParams] = useState<UniverseParameters>({
    darkMatter: 84,
    agnFeedback: 2,
    lowMassStars: 30,
    mediumMassStars: 50,
    highMassStars: 20,
  });

  const handleParamChange = (name: keyof UniverseParameters, value: number) => {
    setParams(prev => ({ ...prev, [name]: value }));
  };
  
  const handleOriginChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    setOriginStory(e.target.value as OriginStory);
  };

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    onSimulate(originStory, params);
  };

  const selectedOriginDescription = ORIGIN_STORIES.find(o => o.value === originStory)?.description;

  return (
    <div className="bg-slate-800/50 p-6 rounded-lg border border-slate-700 shadow-2xl shadow-slate-950">
      <h2 className="text-xl font-bold mb-4 font-orbitron text-cyan-300">Universe Configuration</h2>
      <form onSubmit={handleSubmit}>
        <div className="space-y-6">
          <div>
            <label htmlFor="originStory" className="block text-sm font-medium text-slate-300 mb-1">Origin Story</label>
            <div className="relative">
              <select
                id="originStory"
                value={originStory}
                onChange={handleOriginChange}
                className="w-full bg-slate-700 border border-slate-600 rounded-md py-2 px-3 text-white appearance-none focus:outline-none focus:ring-2 focus:ring-cyan-500"

            </div>
            {selectedOriginDescription && <p className="text-xs text-slate-400 mt-2">{selectedOriginDescription}</p>}
          </div>
          
          <Slider
            label="Dark Matter"
            name="darkMatter"
            value={params.darkMatter}
            onChange={v => handleParamChange('darkMatter', v)}
            min={10} max={95} step={1} unit="%"
            description="The scaffolding of the cosmos. Higher values lead to more galaxy formation."
          />

          <Slider

            name="agnFeedback"

            onChange={v => handleParamChange('agnFeedback', v)}
